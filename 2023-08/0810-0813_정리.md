import sys
from collections import deque
input = sys.stdin.readline

d = [(0, -1), (1, 0), (0, 1), (-1, 0)]

m, n, k = map(int, input().split())
graph = [[0] * m for _ in range(n)]

for _ in range(k):
  x1, y1, x2, y2 = map(int, input().split())
  for i in range(x1, x2):
    for j in range(y1, y2):
      graph[i][j] = 1

result = []

def bfs(x, y):
  q = deque()
  q.append((x, y))
  graph[x][y] = 1
  count = 1

  while q:
    a, b = q.popleft()

    for dx, dy in d:
      nx = a + dx
      ny = b + dy

      if not (0 <= nx < n) or not ( 0 <= ny < m):
        continue
      if graph[nx][ny] == 0:
        graph[nx][ny] = graph[a][b] + 1
        count += 1
        q.append((nx, ny))

  result.append(count)

total = 0
for i in range(n):
  for j in range(m):
    if graph[i][j] == 0:
      total += 1
      bfs(i, j)

print(total)
result.sort()
for i in result:
  print(i, end=' ')
